mkdir -p pcie_pm_dump

sudo lspci -tv > pcie_pm_dump/topology.txt
sudo lspci -nn > pcie_pm_dump/lspci_nn.txt

sudo lspci -vvv -s 0000:02:00.0 > pcie_pm_dump/ep_vvv.txt
sudo lspci -vvv -s 0000:00:01.1 > pcie_pm_dump/rp_vvv.txt

sudo lspci -xxxx -s 0000:02:00.0 > pcie_pm_dump/ep_cfg_xxxx.txt
sudo lspci -xxxx -s 0000:00:01.1 > pcie_pm_dump/rp_cfg_xxxx.txt

cat /proc/cmdline > pcie_pm_dump/cmdline.txt
uname -a > pcie_pm_dump/uname.txt
cat /sys/module/pcie_aspm/parameters/policy > pcie_pm_dump/aspm_policy.txt

cat /sys/class/nvme/nvme0/address > pcie_pm_dump/nvme_address.txt
cat /sys/class/nvme/nvme0/state > pcie_pm_dump/nvme_state.txt
cat /sys/class/nvme/nvme0/device/power/control > pcie_pm_dump/runtime_pm_control.txt 2>/dev/null
cat /sys/class/nvme/nvme0/device/power/runtime_status > pcie_pm_dump/runtime_pm_status.txt 2>/dev/null

nvme list > pcie_pm_dump/nvme_list.txt
nvme id-ctrl /dev/nvme0 > pcie_pm_dump/id_ctrl.txt
nvme get-feature /dev/nvme0 -f 0x0c > pcie_pm_dump/apst.txt 2>&1
nvme get-feature /dev/nvme0 -f 0x02 > pcie_pm_dump/power_state.txt 2>&1

dmesg -T | tail -n 300 > pcie_pm_dump/dmesg_tail.txt
